def initialize(nfil, ncol, h = {})
        super(nfil, ncol)
        @pos = Hash.new({})
        for k in h.keys do
        @pos[k] = if h[k].is_a? VectorDisperso
                                        h[k]
                                 else 
                                        @pos[k] = VectorDisperso.new(h[k])
             end
        end
end

def col(j)
        c = {}
        for r in @pos.keys do
                c[r] = @pos[r].vector[j] if @pos[r].vector.keys.include? j
        end
        VectorDisperso.new c
end
